class Solution {
public:
    int maxProfit(vector<int>& prices) {
        int minprice=prices[0];
        int maxprofit=0;
        int n=prices.size();

        for(int i=0;i<n;i++){
           int profit=prices[i]-minprice;
            maxprofit=max(maxprofit, profit);
            minprice=min(prices[i], minprice);
        }
        return maxprofit;
    }

};
